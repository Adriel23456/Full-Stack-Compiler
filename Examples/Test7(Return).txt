# Función matemática con múltiples tipos
(int) base, altura, resultado;
(bool) esPositivo;
(color) colorFigura;

function calcularArea(base, altura) {
    resultado = base * altura;
    return resultado;
}

frame {
    base = 10;
    altura = 5;
    resultado = calcularArea(base, altura);
    
    if (resultado > 0) {
        esPositivo = true;
        colorFigura = verde;
    } else {
        esPositivo = false;
        colorFigura = rojo;
    }
    
    setcolor(colorFigura);
    draw rect(100, 100, base * 10, altura * 10);
}
