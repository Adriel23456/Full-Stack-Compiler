# Definir variables para la espiral
(color) c;
(int) x, y, t;
clear();

frame {
    loop (t = 0; t < (360*2); t = t + 5) {
        # Calcular coordenadas de la espiral usando trigonometría
        x = 320 + t * cos(t * 3.1416 / 180);  # Centro en X=320
        y = 240 + t * sin(t * 3.1416 / 180);  # Centro en Y=240

        # Cambiar color en cada iteración
        if (t % 3 == 0) { c = rojo; }
        else if (t % 3 == 1) { c = azul; }
        else { c = verde; }

        setcolor(c);
        draw pixel(x, y);
        draw pixel(x+1, y+1);
        draw pixel(x+1, y);
        draw pixel(x, y+1);
        draw pixel(x-1, y-1);
        draw pixel(x-1, y);
        draw pixel(x, y-1);

        wait(100);  # Pequeño retraso para animación suave
    }
}
